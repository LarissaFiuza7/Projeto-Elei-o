@startuml
actor "Sistema Externo" as SE
participant "UEg: Gerenciamento" as UEg
participant "Banco de Dados" as BD

SE -> UEg : Solicita cadastro de nova UEv (dados de identificação)
UEg -> BD : Verifica se UEv já existe (ID)
BD --> UEg : Retorna existe/não existe
alt UEv não existe
    UEg -> BD : Salva dados da nova UEv
    BD --> UEg : Confirmação de cadastro
    UEg -> SE : Cadastro realizado com sucesso
else UEv já existe
    UEg -> SE : Falha (UEv já cadastrada)
end
@enduml
